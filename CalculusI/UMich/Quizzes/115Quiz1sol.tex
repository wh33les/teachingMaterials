\item \textbf{Computations/Algebra.} (1 pt each) 
\begin{enumerate}
\item If $m(z)=z^2$, simplify $m(z+h)-m(z)$.
\begin{eqnarray*}
m(z+h)-m(h) &=& (z+h)^2-z^2 \\
&=& z^2+2zh+h^2-z^2 \\
&=& 2zh +z^2 
\end{eqnarray*}
\vspace{.5pc}
\item Convert the angle $\frac{\pi }{6}$ to degrees.
\begin{eqnarray*}
 \text{degrees} &=& \frac{\pi }{6}\cdot \frac{180 \text{ degrees}}{\pi \text{ radians}} \\
&=& \frac{180}{6} \\
&=& 30
\end{eqnarray*}


\end{enumerate}
