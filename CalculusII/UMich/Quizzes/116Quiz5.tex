\item \textbf{Questions/Problems.}

\[\frac{dv}{dt}=10-\frac{1}{10}(1+e^{-t})v^2.\]
\begin{enumerate}
\item (3 pts)

\hspace{10ex} & \hspace{10ex} & \hspace{10ex} & \hspace{10ex} \\ [0.5ex]
\hline 
\hspace{10ex} & \hspace{10ex} & \hspace{10ex} & \hspace{10ex} \\ [0.5ex]

\end{array}\]
